pr1 = 1


def dec_print(ofunc):
    def wrapper_func(*args, pr):
      if pr <= pr1:
        return ofunc(*args,)
     
     

    return wrapper_func

@dec_print
def display(message):
  print(message)

display('hello5', pr = 5)
display('hello4', pr = 4)
display('hello3', pr = 3)
display('hello2', pr = 2)
display('hello1', pr = 1)
